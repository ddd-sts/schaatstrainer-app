        function t(key) {

        // Language Management

        function setLanguage(lang) {

            currentLanguage = lang;

            localStorage.setItem('schaats_language', lang);


            // Update all elements with data-nl/data-en attributes

            document.querySelectorAll('[data-nl]').forEach(el => {

                const value = el.getAttribute(`data-${lang}`);

                if (value) {

                    if (el.tagName === 'INPUT' || el.tagName === 'TEXTAREA') {

                        if (el.getAttribute('placeholder') !== null) {

                            el.setAttribute('placeholder', value);

                        }

                    } else {

                        el.textContent = value;

                    }

                }

            });


            // Update language button states

            document.querySelectorAll('.lang-btn').forEach(btn => {

                btn.classList.toggle('active', btn.dataset.lang === lang);

            });


            updateConnectionStatus();

        }


        // Connection Status Management

        function updateConnectionStatus(status = null) {

            const statusEl = document.getElementById('connectionStatus');

            if (!statusEl) return;


            let statusText, className;

            if (status === 'connecting') {

                statusText = t('connecting');

                className = 'loading';

            } else if (isConnected) {

                statusText = t('connected');

                className = 'connected';

            } else {

                statusText = t('disconnected');

                className = 'disconnected';

            }


            statusEl.textContent = statusText;

            statusEl.className = `connection-status ${className}`;

        }


        // API Communication

        async function apiCall(action, data = {}) {

            if (!CONFIG.apiEndpoint) {

                throw new Error(t('configRequired'));

            }


            updateConnectionStatus('connecting');

            

            const payload = {

                action: action,

                data: data,

                timestamp: new Date().toISOString(),

                user: currentUser,

                role: currentRole

            };


            let lastError;

            for (let attempt = 1; attempt <= CONFIG.retryAttempts; attempt++) {

                try {

                    const controller = new AbortController();

                    const timeoutId = setTimeout(() => controller.abort(), CONFIG.timeout);


                    const response = await fetch(CONFIG.apiEndpoint, {

                        method: 'POST',

                        headers: {

                            'Content-Type': 'application/json',

                        },

                        body: JSON.stringify(payload),

                        signal: controller.signal

                    });


                    clearTimeout(timeoutId);


                    if (!response.ok) {

                        throw new Error(`HTTP ${response.status}: ${response.statusText}`);

                    }


                    const result = await response.json();

                    

                    isConnected = true;

                    updateConnectionStatus();

                    

                    if (result.error) {

                        throw new Error(result.error);

                    }

                    

                    return result;


                } catch (error) {

                    lastError = error;

                    console.warn(`API call attempt ${attempt} failed:`, error.message);

                    

                    if (attempt < CONFIG.retryAttempts) {

                        await new Promise(resolve => setTimeout(resolve, 1000 * attempt));

                    }

                }

            }


            isConnected = false;

            updateConnectionStatus();

            throw lastError;

        }

        // Authentication

        async function login() {

            const username = document.getElementById('username').value.trim();

            const password = document.getElementById('password').value.trim();

            

            if (!username || !password) {

                showNotification('Vul gebruikersnaam en wachtwoord in', 'warning');

                return;

            }


            if (!CONFIG.apiEndpoint) {

                showNotification(t('configRequired'), 'error');

                showSetup();

                return;

            }


            const loginBtn = document.getElementById('loginBtn');

            loginBtn.disabled = true;

            loginBtn.innerHTML = '<div class="loading"></div> ' + t('connecting');


            try {

                const result = await apiCall('login', { username, password });

                

                if (result.success) {

                    currentUser = username;

                    currentRole = result.role;

                    

                    // Update UI

                    document.getElementById('currentUser').textContent = result.name || username;

                    document.getElementById('userRole').textContent = currentRole.toUpperCase();

                    document.getElementById('userRole').className = `role-badge ${currentRole}`;

                    

                    // Show appropriate interface

                    document.getElementById('loginScreen').classList.add('hidden');

                    document.getElementById('mainApp').classList.remove('hidden');

                    document.getElementById('setupWarning').style.display = 'none';

                    document.getElementById('configPanel').classList.add('hidden');

                    

                    setupRoleBasedUI();

                    showNotification(t('loginSuccess'), 'success');

                    

                } else {

                    throw new Error(result.message || t('loginFailed'));

                }


            } catch (error) {

                console.error('Login failed:', error);

                showNotification(t('loginFailed') + ': ' + error.message, 'error');

            }


            loginBtn.disabled = false;

            loginBtn.innerHTML = '<span data-nl="Inloggen" data-en="Login">Inloggen</span>';

            setLanguage(currentLanguage); // Refresh translations

        }

        // Role-based UI Setup

        function setupRoleBasedUI() {

            const adminTabs = document.querySelectorAll('.admin-only');

            const trainerTabs = document.querySelectorAll('.trainer-only');

            

            if (currentRole === 'admin') {

                adminTabs.forEach(tab => tab.classList.remove('hidden'));

                showTab('dashboard', adminTabs[0]);

            } else {

                adminTabs.forEach(tab => tab.classList.add('hidden'));

                trainerTabs.forEach(tab => tab.classList.remove('hidden'));

                showTab('registration', trainerTabs[0]);

            }

        }


        // Tab Management

        function showTab(tabName, tabElement) {

            // Hide all tab contents

            document.querySelectorAll('.tab-content').forEach(tab => {

                tab.classList.add('hidden');

            });

            

            // Show selected tab

            document.getElementById(tabName).classList.remove('hidden');

            

            // Update tab styling

            document.querySelectorAll('.tab').forEach(tab => {

                tab.classList.remove('active');

            });

            if (tabElement) tabElement.classList.add('active');

            

            // Load tab data

            loadTabData(tabName);

        }


        async function loadTabData(tabName) {

            if (!isConnected) {

                showNotification(t('noConnection'), 'warning');

                return;

            }


            try {

                switch (tabName) {

                    case 'dashboard':

                        await loadDashboard();

                        break;

                    case 'userManagement':

                        await loadUsers();

                        break;

                    case 'masterData':

                        await loadAttendees();

                        break;

                    case 'registration':

                        await loadRegistrationData();

                        break;

                    case 'overview':

                        await loadTrainerOverview();

                        break;

                    case 'invoicing':

                        await loadInvoicingData();

                        break;

                    case 'settings':

                        await loadTrainerSettings();

                        break;

                }

            } catch (error) {

                console.error('Failed to load tab data:', error);

                showNotification('Fout bij laden gegevens: ' + error.message, 'error');

            }

        }

        function toggleAttendee(attendeeId) {

            const attendee = allAttendees.find(a => a.id === attendeeId);

            if (!attendee) return;

            

            const existingIndex = selectedAttendees.findIndex(a => a.id === attendeeId);

            

            if (existingIndex > -1) {

                selectedAttendees.splice(existingIndex, 1);

            } else {

                selectedAttendees.push(attendee);

            }

            

            displayAttendeesGrid();

        }

        async function registerSession() {

            const date = document.getElementById('trainingDate').value;

            const startTime = document.getElementById('trainingStartTime').value;

            const endTime = document.getElementById('trainingEndTime').value;

            if (!date || !startTime || !endTime || selectedAttendees.length === 0) {

                showNotification('Vul alle verplichte velden in en selecteer deelnemers', 'warning');

                    startTime,

                    endTime,

                    type,

        function showNotification(message, type = 'success') {

            const existingNotifications = document.querySelectorAll('.notification');

            existingNotifications.forEach(notification => notification.remove());

            

            const notification = document.createElement('div');

            notification.textContent = message;

            

            document.body.appendChild(notification);

            

            setTimeout(() => {

                notification.style.animation = 'slideOut 0.3s ease-in';

                setTimeout(() => {

                    if (notification.parentNode) {

                        notification.parentNode.removeChild(notification);

                    }

                }, 300);

            }, 3000);

        }
